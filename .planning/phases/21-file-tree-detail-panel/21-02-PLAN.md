On mount effect — fetch root entries:
```typescript
$effect(() => {
  rootLoading = true;
  fetchTree(repoName).then(r => {
    rootEntries = r.entries;
    rootLoading = false;
  }).catch(() => { rootLoading = false; });
});
```

NOTE: The `$effect` above reads `repoName`, so it re-runs when repoName changes (user switches repo tabs). Reset all state when repoName changes by setting rootEntries to `[]`, children to `{}`, expanded to `{}`.

Toggle function:
```typescript
async function handleToggle(entry: TreeEntry) {
  if (entry.isDir) {
    if (expanded[entry.path]) {
      // Collapse
      expanded[entry.path] = false;
    } else {
      // Expand — fetch children if not cached
      if (!children[entry.path]) {
        loading[entry.path] = true;
        try {
          const result = await fetchTree(repoName, entry.path);
          children[entry.path] = result.entries;
        } catch { /* leave empty */ }
        loading[entry.path] = false;
      }
      expanded[entry.path] = true;
    }
    // Also select the directory for detail panel
    onSelectDir(entry.path);
  } else {
    onSelectFile(entry.path);
  }
}
```

Auto-expand to selectedPath effect — when selectedPath changes (e.g., from dependency link click or URL hash navigation), expand all parent directories:
```typescript
let lastExpandedFor: string | null = $state(null);

$effect(() => {
  if (!selectedPath || selectedPath === lastExpandedFor) return;
  lastExpandedFor = selectedPath;
  const parts = selectedPath.split('/');
  for (let i = 1; i < parts.length; i++) {
    const prefix = parts.slice(0, i).join('/');
    if (!expanded[prefix]) {
      // Trigger expansion (fetch children if needed)
      expanded[prefix] = true;
      if (!children[prefix]) {
        fetchTree(repoName, prefix).then(r => {
          children[prefix] = r.entries;
        });
      }
    }
  }
});
```

Recursive render function — use a `renderEntries` approach. For each entry in an array, render a `FileTreeNode` and if it's an expanded directory with loaded children, recursively render its children at depth+1.

Template:
```svelte
<div class="overflow-y-auto h-full py-1">
  {#if rootLoading}
    <p class="text-gray-500 text-sm px-4 py-2">Loading tree...</p>
  {:else if rootEntries.length === 0}
    <p class="text-gray-500 text-sm px-4 py-2">No files found.</p>
  {:else}
    {#each rootEntries as entry (entry.path)}
      <FileTreeNode
        {entry}
        depth={0}
        isExpanded={!!expanded[entry.path]}
        isSelected={selectedPath === entry.path}
        isLoading={!!loading[entry.path]}
        onToggle={handleToggle}
      />
      {#if expanded[entry.path] && children[entry.path]}
        <!-- Recursive rendering of children -->
        {#each children[entry.path] as child (child.path)}
          <FileTreeNode
            entry={child}
            depth={1}
            isExpanded={!!expanded[child.path]}
            isSelected={selectedPath === child.path}
            isLoading={!!loading[child.path]}
            onToggle={handleToggle}
          />
          <!-- Continue nesting for deeper levels... -->
        {/each}
      {/if}
    {/each}
  {/if}
</div>
```

IMPORTANT: The naive nested `{#each}` only handles 2 levels. For arbitrary depth, use a Svelte 5 snippet with `{@render}` for recursion:

```svelte
{#snippet renderEntries(entries: TreeEntry[], depth: number)}
  {#each entries as entry (entry.path)}
    <FileTreeNode
      {entry}
      {depth}
      isExpanded={!!expanded[entry.path]}
      isSelected={selectedPath === entry.path}
      isLoading={!!loading[entry.path]}
      onToggle={handleToggle}
    />
    {#if entry.isDir && expanded[entry.path] && children[entry.path]}
      {@render renderEntries(children[entry.path], depth + 1)}
    {/if}
  {/each}
{/snippet}

<div class="overflow-y-auto h-full py-1">
  {#if rootLoading}
    <p class="text-gray-500 text-sm px-4 py-2">Loading tree...</p>
  {:else if rootEntries.length === 0}
    <p class="text-gray-500 text-sm px-4 py-2">No files found.</p>
  {:else}
    {@render renderEntries(rootEntries, 0)}
  {/if}
</div>
```

```typescript
let treeWidth = $state(30); // percent

function onDividerPointerDown(e: PointerEvent) {
  const container = (e.currentTarget as HTMLElement).parentElement!;
  const startX = e.clientX;
  const startWidth = treeWidth;

  function onMove(ev: PointerEvent) {
    const delta = ev.clientX - startX;
    const containerWidth = container.getBoundingClientRect().width;
    const newPct = startWidth + (delta / containerWidth) * 100;
    treeWidth = Math.max(15, Math.min(70, newPct)); // clamp 15%–70%
  }
  function onUp() {
    document.removeEventListener('pointermove', onMove);
    document.removeEventListener('pointerup', onUp);
  }
  document.addEventListener('pointermove', onMove);
  document.addEventListener('pointerup', onUp);
  e.preventDefault();
}
```
