/**

export function getStaleCount(db: InstanceType<typeof Database>): number {
  const row = db.prepare(
    'SELECT COUNT(*) AS n FROM files WHERE is_directory = 0 AND (summary_stale_since IS NOT NULL OR concepts_stale_since IS NOT NULL OR change_impact_stale_since IS NOT NULL)'
  ).get() as { n: number };
  return row.n;
}

Add a `dotColor` function in the `<script>` block after the existing helper functions:

```typescript
function dotColor(repo: RepoListItem): string {
  if (!repo.online) return 'bg-gray-500';
  const FIVE_MIN = 5 * 60 * 1000;
  const isRecent = repo.dbMtimeMs != null && (Date.now() - repo.dbMtimeMs) < FIVE_MIN;
  if (isRecent) return 'bg-green-500';
  if (repo.staleCount > 0) return 'bg-orange-400';
  return 'bg-gray-500';
}
```

**2. Add periodic repo polling to App.svelte:**

Replace the current one-shot `$effect` that calls `fetchRepos()` with one that also sets up a 30-second polling interval:

```typescript
$effect(() => {
  let cancelled = false;

  function loadRepos() {
    fetchRepos().then(r => {
      if (cancelled) return;
      repos = r;
      loading = false;
      // Navigate to first repo on initial load only
      if ((hash === '#/' || hash === '#') && r.length > 0) {
        window.location.hash = `#/project/${r[0].name}`;
      }
    }).catch(err => {
      if (cancelled) return;
      console.error('Failed to load repos:', err);
      loading = false;
    });
  }

  loadRepos();
  const interval = setInterval(loadRepos, 30_000);

  return () => {
    cancelled = true;
    clearInterval(interval);
  };
});
```

The `cancelled` flag prevents state updates after the effect is cleaned up. The 30s interval keeps tab status dots current without heavy polling overhead (D-08). The initial-load auto-navigation to the first repo is preserved.
