```typescript
export interface RicherEdgeData {
  /** Raw import specifiers from import_statement and require() nodes */
  regularImports: string[];
  /** Raw specifiers from export_statement nodes that have a source field (re-exports) */
  reExportSources: string[];
  /** className + sourceSpecifier pairs from extends_clause where class is imported from another file */
  inheritsFrom: Array<{ className: string; sourceSpecifier: string }>;
}
```

Add the `extractRicherEdges()` function. This function:
1. Parses the file using `getParser(filePath)` (the existing internal function)
2. Walks the AST in a single pass collecting three categories:
   - `regularImports`: from `import_statement` source field + `call_expression` with `require()`
   - `reExportSources`: from `export_statement` nodes that have BOTH a `source` field (re-export from another module)
   - `inheritsFrom`: from `class_declaration` nodes with heritage/extends clause, cross-referenced with an import name map

Implementation:

```typescript
export function extractRicherEdges(filePath: string, source: string): RicherEdgeData | null {
  const parser = getParser(filePath);
  if (!parser) return null;

  let tree;
  try {
    tree = parser.parse(source);
  } catch (err) {
    log(`[ast-parser] tree-sitter parse failed for ${filePath}: ${err}`);
    return null;
  }

  const regularImports: string[] = [];
  const reExportSources: string[] = [];
  const inheritsFrom: Array<{ className: string; sourceSpecifier: string }> = [];

  // Phase 1: Build import name → source specifier map for inherits correlation
  // Also collect regular imports and re-export sources
  const importNameToSource = new Map<string, string>();

  function visitNode(node: any): void {
    if (node.type === 'import_statement') {
      const sourceNode = node.childForFieldName('source');
      if (sourceNode) {
        const specifier = getStringFragment(sourceNode);
        if (specifier) {
          regularImports.push(specifier);
          // Build name map: for "import { Foo, Bar } from './mod'"
          // Look for import_clause -> named_imports -> import_specifier children
          buildImportNameMap(node, specifier, importNameToSource);
        }
      }
    } else if (node.type === 'export_statement') {
      const sourceNode = node.childForFieldName('source');
      if (sourceNode) {
        const specifier = getStringFragment(sourceNode);
        if (specifier) {
          reExportSources.push(specifier);
        }
      }
      // Don't return early -- still need to recurse for nested nodes
    } else if (node.type === 'call_expression') {
      const fnNode = node.childForFieldName('function');
      if (fnNode && fnNode.type === 'identifier' && fnNode.text === 'require') {
        const argsNode = node.childForFieldName('arguments');
        if (argsNode && argsNode.childCount >= 3) {
          const firstArg = argsNode.child(1);
          if (firstArg && firstArg.type === 'string') {
            const fragment = getStringFragment(firstArg);
            if (fragment) regularImports.push(fragment);
          }
        }
      } else if (fnNode && fnNode.type === 'import') {
        const argsNode = node.childForFieldName('arguments');
        if (argsNode && argsNode.childCount >= 3) {
          const firstArg = argsNode.child(1);
          if (firstArg && firstArg.type === 'string') {
            const fragment = getStringFragment(firstArg);
            if (fragment) regularImports.push(fragment);
          }
        }
      }
    }

    // Phase 2 inline: detect class extends
    if (node.type === 'class_declaration' || node.type === 'class') {
      // Look for extends clause in named children
      for (let i = 0; i < node.namedChildCount; i++) {
        const child = node.namedChild(i);
        if (child.type === 'class_heritage') {
          // class_heritage contains extends_clause children
          for (let j = 0; j < child.namedChildCount; j++) {
            const hChild = child.namedChild(j);
            if (hChild.type === 'extends_clause') {
              // Try childForFieldName('value') first, then fall back to first named child
              const valueNode = hChild.childForFieldName('value') ?? hChild.namedChild(0);
              if (valueNode) {
                const className = valueNode.text as string;
                const sourceSpec = importNameToSource.get(className);
                if (sourceSpec) {
                  inheritsFrom.push({ className, sourceSpecifier: sourceSpec });
                }
              }
            }
          }
        }
      }
    }

    for (let i = 0; i < node.childCount; i++) visitNode(node.child(i));
  }

  visitNode(tree.rootNode);
  return { regularImports, reExportSources, inheritsFrom };
}
```

Add the `buildImportNameMap` helper (internal, not exported):
```typescript
function buildImportNameMap(importNode: any, sourceSpecifier: string, map: Map<string, string>): void {
  // Walk the import statement's children looking for import specifiers
  // Handles: import { Foo } from './mod', import Foo from './mod', import * as Foo from './mod'
  function walk(node: any): void {
    if (node.type === 'import_specifier') {
      // Named import: import { Foo } from './mod'
      const nameNode = node.childForFieldName('name');
      if (nameNode) map.set(nameNode.text as string, sourceSpecifier);
    } else if (node.type === 'identifier' && node.parent?.type === 'import_clause') {
      // Default import: import Foo from './mod'
      map.set(node.text as string, sourceSpecifier);
    } else if (node.type === 'namespace_import') {
      // Namespace import: import * as Foo from './mod'
      const nameNode = node.childForFieldName('name');
      if (nameNode) map.set(nameNode.text as string, sourceSpecifier);
    }
    for (let i = 0; i < node.childCount; i++) walk(node.child(i));
  }
  walk(importNode);
}
```

```typescript
async function extractTsJsEdges(filePath, content, projectRoot): Promise<EdgeResult[]> {
  const richer = extractRicherEdges(filePath, content);
  if (!richer) return []; // parse failed

  const edges: EdgeResult[] = [];

  // Regular imports
  for (const imp of richer.regularImports) {
    const edge = await resolveTsJsImport(imp, filePath, projectRoot, 'imports');
    if (edge) edges.push(edge);
  }

  // Re-exports
  for (const src of richer.reExportSources) {
    const edge = await resolveTsJsImport(src, filePath, projectRoot, 're_exports');
    if (edge) edges.push(edge);
  }

  // Inherits (cross-file class extends)
  for (const { sourceSpecifier } of richer.inheritsFrom) {
    const edge = await resolveTsJsImport(sourceSpecifier, filePath, projectRoot, 'inherits');
    if (edge) edges.push(edge);
  }

  return edges;
}
```

**Part C: Add weight aggregation to `extractEdges()` in language-config.ts**

Modify the `extractEdges()` public function to aggregate duplicate edges by summing weights AFTER the extractor returns raw edges. Per D-12, this aggregation belongs in `extractEdges()`, not in individual extractors.

Replace the simple `return await config.extract(...)` with:

```typescript
export async function extractEdges(
  filePath: string,
  content: string,
  projectRoot: string
): Promise<EdgeResult[]> {
  ensureRegexExtractors();
  const ext = path.extname(filePath).toLowerCase();
  const config = registry.get(ext);
  if (!config) return [];

  let rawEdges: EdgeResult[];
  try {
    rawEdges = await config.extract(filePath, content, projectRoot);
  } catch (err) {
    log(`[language-config] extractEdges failed for ${filePath}: ${err}`);
    return [];
  }

  // Aggregate duplicate targets by summing weights (D-11/D-12)
  // Key: target + edgeType (same file can be imported AND re-exported — different edges)
  const accumulator = new Map<string, EdgeResult>();
  for (const edge of rawEdges) {
    const key = `${edge.target}\x00${edge.edgeType}`;
    const existing = accumulator.get(key);
    if (existing) {
      existing.weight += edge.weight;
    } else {
      accumulator.set(key, { ...edge });
    }
  }
  return Array.from(accumulator.values());
}
```

The aggregation key MUST include `edgeType` per Pitfall 5. A file that is both imported and re-exported produces TWO distinct edges, not one merged edge.
